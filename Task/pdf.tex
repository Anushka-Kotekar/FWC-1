\section*{1.}
Find the value of
\[
\begin{vmatrix}
a-b & b-c & c-a \\
b-c & c-a & a-b \\
c-a & a-b & b-c
\end{vmatrix}
\]

\section*{2.}
If
\[
A=\begin{pmatrix}
1 & -2 & 3\\
-4 & 2 & 5
\end{pmatrix},
\quad
B=\begin{pmatrix}
2 & 3\\
4 & 5\\
2 & 1
\end{pmatrix}
\]
and \(BA=(b_{ij})\), find \(b_{21}+b_{32}\).

\section*{3.}
Write the number of all possible matrices of order \(2\times3\) with each entry equal to 1 or 2.

\section*{4.}
Write the coordinates of the reflection of \((\alpha,\beta,\gamma)\) in the XZ-plane.

\section*{5.}
Find the position vector dividing internally in the ratio \(1:3\) the join of
\[
\vec{a}+3\vec{b} \quad \text{and} \quad \vec{a}-\vec{b}.
\]

\section*{6.}
If \(|\vec{a}|=4\), \(|\vec{b}|=3\) and \(\vec{a}\cdot\vec{b}=6\sqrt{3}\),
find \(|\vec{a}\times\vec{b}|\).

\section*{7.}
Solve for \(x>0\):
\[
\arctan\!\left(\frac{2-x}{2+x}\right)
=
\frac{1}{2}\arctan\!\left(\frac{x}{2}\right).
\]

\textbf{OR}

Prove that
\[
2\sin^{-1}\!\left(\frac{3}{5}\right)
-
\arctan\!\left(\frac{17}{31}\right)
=
\frac{\pi}{4}.
\]

\section*{8.}
In an orphanage, an amount of money was distributed among the children.

If there were 8 children less, each child would have received Rs.\ 10 more.  
If there were 16 children more, each child would have received Rs.\ 10 less.

Using the matrix method, find the number of children and the total amount distributed.
What values are reflected by this decision?

\section*{9.}
If \(x=e^{\cos 2t}\) and \(y=e^{\sin 2t}\), prove that
\[
\frac{dy}{dx}
=
-\frac{y\log x}{x\log y}.
\]

\textbf{OR}

Verify the Mean Value Theorem for the function
\[
f(x)=2\sin x+\sin 2x
\quad \text{on} \quad [0,\pi].
\]

\section*{10.}
Show that the function \(f\) defined by
\[
f(x)=
\begin{cases}
\dfrac{e^{1/x}-1}{e^{1/x}+1}, & x\neq0,\\
-1, & x=0
\end{cases}
\]
is discontinuous at \(x=0\).

\section*{11.}
Find the equation of the tangent to the curve
\[
y=\sqrt{5x-3}-5
\]
which is parallel to the line
\[
4x-2y+5=0.
\]

\section*{12.}
Evaluate:
\[
\int_{1}^{5}\left(|x-1|+|x-2|+|x-3|\right)\,dx.
\]

\textbf{OR}

Evaluate:
\[
\int_{0}^{\pi}\frac{x\sin x}{1+3\cos^2x}\,dx.
\]

\section*{13.}
Evaluate:
\[
\int\frac{2x+1}{(x^2+1)(x^2+4)}\,dx.
\]
